public static double[] HoareSort(double[] array, int left, int right) { int base, opposite, p; double c; base = left; opposite = right; while (base != opposite) { if ((array[base] > array[opposite]) ^ (base > opposite)) { c = array[base]; array[base] = array[opposite]; array[opposite] = c; p = base; base = opposite; if (p < opposite) opposite = p + 1; else opposite = p - 1; } else { if (base < opposite) opposite--; else opposite++; } } if (left < base - 1) HoareSort(array, left, base - 1); if (base + 1 < right) HoareSort(array, base + 1, right); return array; } }
